# Scalar implementation of Submanifold Sparse Convolution

- Input : 3x5x5x5 (channel x depth x height x width)
- Kernel : 4x3x3x3x3 (out_channel x in_channel x Kernel_size)
- Padding : 1x1x1
- Padded Input : 3x7x7x7
- <b>output: 4x5x5x5</b>

## My Implementation 
```c++
#include<iostream>
#include<vector>
#include "c7x.h"
#include "c7x_scalable.h"
#include <cstdlib> 
#include <ctime>
#include<chrono>


#include <stdio.h>
#include <sys/time.h>
 
#define CPU_FREQ 2e9
#include <unordered_map>
#include <vector>

using namespace c7x;
using namespace std;

int main(){
    int32_t h=5,w=5,d=5,ch_i=3,ch_o=4,k_s=3;
    int32_t pad=1;
    int32_t inp[ch_i][d][h][w];
    int32_t pad_d=d+(2*pad), pad_w=w+(2*pad), pad_h=h+(2*pad);
    int32_t pad_inp[ch_i][pad_d][pad_h][pad_w];
    int32_t out[ch_o][d][h][w];
    int32_t ker[ch_o][ch_i][k_s][k_s][k_s];

    unordered_map<int, vector<int>> table;

    //Input
    for(int32_t i=0; i<ch_i; i++){
        for(int32_t j=0; j<d; j++){
            for(int32_t k=0; k<h; k++){
                for(int32_t m=0; m<w; m++){
                    inp[i][j][k][m] =0;

                  
                }
            }
        }
    }

    int32_t count=0;
    for(int32_t i=0; i<ch_i; i++){
        for(int32_t j=0; j<d; j++){
            for(int32_t k=0; k<h; k++){
                for(int32_t m=0; m<w; m++){
                    if(m==0 && k==0 ){
                        inp[i][j][k][m] = count++;

                    }
                }
            }
        }
    }
    //Initilizing zeross to Padded Input
    for(int32_t i=0; i<ch_i; i++){
        for(int32_t j=0; j<pad_d; j++){
            for(int32_t k=0; k<pad_h; k++){
                for(int32_t m=0; m<pad_w; m++){
                    pad_inp[i][j][k][m] = 0;
                }
            }
        }
    }

    //Padded Input
    for(int32_t i=0; i<ch_i; i++){
        for(int32_t j=pad; j<(d+pad); j++){
            for(int32_t k=pad; k<(h+pad); k++){
                for(int32_t m=pad; m<(w+pad); m++){
                    pad_inp[i][j][k][m] = inp[i][j-pad][k-pad][m-pad];
                }
            }
        }
    }

    //Kernel
    int32_t k_count=0;
    for(int32_t i=0; i<ch_o; i++){
        for(int32_t j=0; j<ch_i; j++){
            k_count=0;
            for(int32_t k=0; k<k_s; k++){
                for(int32_t m=0; m<k_s; m++){
                    for(int32_t n=0; n<k_s; n++){
                        ker[i][j][k][m][n]=k_count++;
                    }
                }
            }
        }
        // k_count++;
    }

    //Find active region from Padded inp
    int32_t id=0;
    for(int32_t i=0; i<ch_i; i++){
        for(int32_t j=0; j<pad_d; j++){
            for(int32_t k=0; k<pad_h; k++){
                for(int32_t m=0; m<pad_w; m++){
                    if(pad_inp[i][j][k][m]!=0){
                        table[id++]={i,j,k,m};
                    }
                    
                }
            }
        }
    }
    int offset = k_s / 2;
    for(auto &[idx, index] : table){
        int16_t channel = index[0], depth=index[1], height=index[2], width=index[3];
        for(int32_t i=0; i<ch_o; i++){
            int32_t sum=0;
            for(int32_t j=0; j<ch_i; j++){
                for(int32_t k=-offset; k<=offset; k++){
                    for(int32_t m=-offset; m<=offset; m++){
                        for(int32_t n=-offset; n<=offset; n++){
                            if(pad_inp[j][depth+k][height+m][width+n]!=0){

                                sum+=ker[channel][j][k+offset][m+offset][n+offset]*pad_inp[j][depth+k][height+m][width+n];
                            }
                        }
                    }
                }
            }
            cout<<"Sum :"<<sum<<"\n";
            
            out[channel][depth-1][height-1][width-1]=sum;
            cout<<"Channel out:"<<i<<", Depth :"<<depth-1<<", Height:"<<height-1<<", Width:"<<width-1<<"Out :"<<out[i][depth-1][height-1][width-1]<<"Cordinate"<<"("<<channel<<depth<<height<<width<<")"<<"\n";
        }

        // cout<<"Channel :"<<channel<<", Depth :"<<depth<<", Height :"<<height<<", width :"<<width<<"\n";

    }


/*

    //Printing
    cout<<"INPUT :"<<"\n";
    for(int32_t i=0; i<ch_i; i++){
        for(int32_t j=0; j<d; j++){
            for(int32_t k=0; k<h; k++){
                for(int32_t m=0; m<w; m++){
                    cout<<inp[i][j][k][m]<<" | ";
                }
                cout<<"\n";
            }
            cout<<"Depth:"<<j<<"\n";
        }
        cout<<"Channel:"<<i<<"\n";
    }
*/

    cout<<"PADDED INPUT :"<<"\n";
    for(int32_t i=0; i<ch_i; i++){
        cout<<"Channel:"<<i<<"\n";
        for(int32_t j=0; j<pad_d; j++){
            cout<<"Depth:"<<j<<"\n";
            for(int32_t k=0; k<pad_h; k++){
                for(int32_t m=0; m<pad_w; m++){
                    cout<<pad_inp[i][j][k][m]<<" | ";
                }
                cout<<"\n";
            }
        }
    }


    cout<<"Kernel:";
    for(int32_t i=0; i<ch_o; i++){
        cout<<"OUT ch:"<<i<<"\n";
        for(int32_t j=0; j<ch_i; j++){
            cout<<"In ch:"<<j<<"\n";
            k_count=0;
            for(int32_t k=0; k<k_s; k++){
                for(int32_t m=0; m<k_s; m++){
                    for(int32_t n=0; n<k_s; n++){
                        cout<<ker[i][j][k][m][n]<<" | ";
                    }
                    cout<<"\n";
                }
                cout<<"\n";
            }
            cout<<"\n";
        }
        cout<<"\n";
    }

    cout<<"OUTPUT :"<<"\n";
    for(int32_t i=0; i<ch_o; i++){
        cout<<"Channel:"<<i<<"\n";
        for(int32_t j=0; j<d; j++){
            cout<<"Depth:"<<j<<"\n";
            for(int32_t k=0; k<h; k++){
                for(int32_t m=0; m<w; m++){
                    cout<<out[i][j][k][m]<<" | ";
                }
                cout<<"\n";
            }
        }
    }
}
```
